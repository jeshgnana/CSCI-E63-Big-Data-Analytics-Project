# Method 1: Try undersampling (reducing) the number of genuine transactions and determine if this improves 
# the specificity (ability to detect truly fraudulent transactions)


# set seed to ensure reproducibility
np.random.seed(1)


# this function is supposed to generate a random split of training indices using np.random.choice with 
# a sample size of "number_in_train"
# convert numpy array to list and then filters the rows in dataframe where respective indices are found. 
# Whatever indices are not in the numpy array are made a part of the test set.
# then return the training and test sets.  

# code adapted from https://stackoverflow.com/questions/40421845/pyspark-dataframe-filter-or-include-based-on-list
def gen_test_train_split(df,number_in_train):
    train_sample=np.random.choice(df.select('rowID').toPandas()['rowID'],number_in_train,replace=False)
    sample_list=train_sample.tolist()
    train=df.filter(df.rowID.isin(sample_list))
    test=df.filter(~df.rowID.isin(sample_list))
    return [train,test]


# this function takes in an iteration number as an input, a training set, a test set, 
# the names of predictors and outcome variables and returns some metrics on the model 
# like sensitivity, specificity, and accuracy. 

def perform_log_regression(iteration,train,test,predictors,outcome_variable):
    lr = LogisticRegression(featuresCol = predictors, labelCol=outcome_variable, maxIter=5, regParam=0.05, elasticNetParam=0.1)
    fit_model=lr.fit(train)
    train_outcome=fit_model.transform(train).select(outcome_variable, "prediction")
    test_outcome=fit_model.transform(test).select(outcome_variable, "prediction")
    accuracy=float((test_outcome.filter(test_outcome.Outcome==test_outcome.prediction).count()))/(test_outcome.count())
    specificity=(test_outcome.filter(test_outcome.Outcome==1).filter(test_outcome.prediction==1).count())/float(test_outcome.filter(test_outcome.Outcome==1).count())
    sensitivity=(test_outcome.filter(test_outcome.Outcome==0).filter(test_outcome.prediction==0).count())/float(test_outcome.filter(test_outcome.Outcome==0).count())
    return [iteration,accuracy,sensitivity,specificity]

# Method 2: Expand the prior algorithm to include a series of different samplings for 
# both the fraudulent and genuine transactions. We will develop different pairs of fraudulent and 
# genuine transactions in the test dataset and compare their performance with sensitivity and specificity.  

# set seed for reproducibility 
np.random.seed(1)

# declare number of iterations
# and the number of fraudulent and genuine samples in each iteration.
nIterations=20
num_fraudulent=[5,10,15]
num_genuine= [10,25,50,100]

# lists to store the metrics (sensitivity, specificity)
store_metrics=[]
store_coefficients=[]


# for a given iteration this function will try different splits of number of genuine transactions 
# and fraudulent transactions and return results. For a given iteration, there will be len(num_fraudulent)
# * len(num_genuine) splits. 

for numIter in np.arange(1,nIterations+1):
    for fraud_size in num_fraudulent:
        for genuine_size in num_genuine:
            fraud_train,fraud_test=gen_test_train_split(fraud,fraud_size)
            genuine_train,genuine_test=gen_test_train_split(genuine,genuine_size)
            train=fraud_train.unionAll(genuine_train)
            test=fraud_test.unionAll(genuine_test)
            iteration,accuracy,sensitivity,specificity=perform_log_regression(iteration,train,test,'features','Outcome')
            store_metrics.append((numIter,genuine_size,fraud_size,accuracy,sensitivity,specificity))
    

# convert lists to dataframe, assign column names, and display the mean and standard error of specific metrics
# after grouping by the num of genuine and fraudulent transactions in the split.
# this should help us identify the best splits. 


store_metrics=pd.DataFrame(store_metrics)
store_metrics.columns=['iterationNum','num_genuine','num_fraudulent','accuracy','sensitivity','specificity']

# Method 3: Use sklearn, which has functionality for generating splits that are more representative of the whole 
# dataset, to generate an additional logistic regression model. 
# Applied documentation from https://scikit-learn.org/stable/modules/generated/sklearn.linear_model.LogisticRegression.html
    

# this function takes in a cost value, a training set of predictors, a training set of out the outcome variable
# a test set of predictors, and a test set of outcomes and generates a logistic regression. 
# then it generates predictions based on model and generates similar metrics to assess the model 
# precision, recall(sensitivity), and specificity. 
# it also returns coefficients of the model to be stored in a list. 
def classify_and_return_results(cost_value,train_pred,train_outcome,test_pred,test_outcome):
    logistic_classifier=LogisticRegression(C=cost_value,fit_intercept=True,dual=False, 
                           penalty='l2',solver='sag', tol=0.005, max_iter=300, 
                             class_weight='balanced', n_jobs=None,verbose=0,intercept_scaling=1.0, 
                             multi_class='multinomial', random_state=None,warm_start=True).fit(train_pred,train_outcome) 
    test_predictions=logistic_classifier.predict(test_pred)
    # the average='weighted' provides the most accurate calculation for imbalanced datasets, hence my logic for using it
    recall_value=recall_score(test_outcome,test_predictions,average='weighted')
    precision_value=precision_score(test_outcome,test_predictions,average='weighted')
    specificity=len(test_predictions[(test_outcome==1) & (test_predictions==1)])/float(len(test_outcome[test_outcome==1]))
    num_variables_in_model=len(logistic_classifier.coef_.flatten())
    coefficients=logistic_classifier.coef_.flatten()
    return cost_value,recall_value,precision_value,specificity,num_variables_in_model,coefficients
